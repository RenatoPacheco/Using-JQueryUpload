$('#com-jqueryUpload').fileupload({
    dataType: 'json',
    dropZone: $('.arraste-aqui'),
    add: function (e, data) {
        // É executado quando um ou mais arquivos são selecionados.
    },
    done: function (e, data) {
        // É executado quando um arquivo termina de ser enviado.
    },
    fail: function (e, data) {         
        // É executado quando ocorre um erro no envio do arquivo.
    },
    progress: function (e, data) {
        // É executado durente o envio do arquivo
        var progress = parseInt(data.loaded / data.total * 100, 10);
        progress = progress + '%';
    },
    progressall: function (e, data) {         
        // É executado durente o envio do arquivo
        var progress = parseInt(data.loaded / data.total * 100, 10);
        progress = progress + '%';            
    }
});
